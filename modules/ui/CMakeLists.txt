PROJECT (MOONSHINE_UI C)

SET(MOONSHINE_UI_BINARY_DIR "${PROJECT_BINARY_DIR}/moonshine/ui")

INCLUDE_DIRECTORIES(BEFORE
	${SLANG_INCLUDE_DIR}
	${GLIB_INCLUDE_DIRS}
	${MOONSHINE_CORE_SOURCE_DIR})
MESSAGE(STATUS ${MOONSHINE_UI_BINARY_DIR})
## Build rules.
FOREACH(lib entry) # buffer statusbar term)
	ADD_LIBRARY(${lib} MODULE ${lib}.c)
	TARGET_LINK_LIBRARIES(${lib} ${GLIB_LIBRARY} ${SLANG_LIBRARY})
	SET_TARGET_PROPERTIES(${lib} PROPERTIES PREFIX "")
ENDFOREACH()
